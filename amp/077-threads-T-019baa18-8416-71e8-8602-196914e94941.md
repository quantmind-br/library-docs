#### **Key Models:**

1. [**OrganizationInvite**](https://github.com/PostHog/posthog/blob/master/posthog/models/organization_invite.py) - The invite record
   
   - `organization` (FK): Which org the invite is for
   - `target_email` (EmailField): Invited email address
   - `created_by` (FK to User): Who sent the invite
   - `level` (PositiveSmallIntegerField): MEMBER, ADMIN, or OWNER
   - `private_project_access` (JSONField): List of `[{"id": team_id, "level": "member"|"admin"}]`
   - `created_at`: Used to determine expiration (72 hours via `INVITE_DAYS_VALIDITY`)
   - `is_member_join_email_enabled`: Flag on organization
2. [**User**](https://github.com/PostHog/posthog/blob/master/posthog/models/user.py#L347-L391) - The user account
   
   - `current_organization`: Cached current org
   - `current_team`: Cached current team
   - `organizations`: M2M through OrganizationMembership
3. [**OrganizationMembership**](https://github.com/PostHog/posthog/blob/master/posthog/models/organization.py#L476-L596) - User's membership in organization
   
   - `organization` (FK)
   - `user` (FK)
   - `level` (MEMBER=1, ADMIN=8, OWNER=15)
   - `joined_at`: Tracks when user joined
4. [**AccessControl**](https://github.com/PostHog/posthog/tree/master/ee/models/rbac) - Project-level permissions
   
   - `team` (FK): The project/team
   - `organization_member` (FK to OrganizationMembership): Who has access
   - `resource`: Always "project"
   - `resource_id`: The team ID
   - `access_level`: "member", "admin", or "none"
5. [**UserProductList**](https://github.com/PostHog/posthog/tree/master/posthog/models/file_system) - User's product preferences
   
   - Synced from colleagues during invite acceptance

### **Detailed View of invite.use() Method**

**Location:** [posthog/models/organization\_invite.py#L125-L162](https://github.com/PostHog/posthog/blob/master/posthog/models/organization_invite.py#L125-L162)

```
def use(self, user: "User", *, prevalidated: bool = False) -> None:
    # Step 1: Validation
    if not prevalidated:
        self.validate(user=user)  # Checks: correct email, not expired, not already member
    
    # Step 2: Create membership (calls User.join())
    user.join(organization=self.organization, level=self.level)

    # Step 3: Create AccessControl for private projects
    for item in self.private_project_access or []:
        try:
            team: Team = self.organization.teams.get(id=item["id"])
            parent_membership = OrganizationMembership.objects.get(
                organization=self.organization,
                user=user,
            )
        except Team.DoesNotExist:
            continue  # Team was deleted, skip

        AccessControl.objects.create(
            team=team,
            resource="project",
            resource_id=str(team.id),
            organization_member=parent_membership,
            access_level=item["level"],
        )

    # Step 4: Sync product preferences
    self._sync_user_product_list_for_accessible_teams(user)

    # Step 5: Send member join email
    if is_email_available(with_absolute_urls=True) and self.organization.is_member_join_email_enabled:
        send_member_join.apply_async(
            kwargs={
                "invitee_uuid": user.uuid,
                "organization_id": self.organization_id,
            }
        )
    
    # Step 6: Delete all invites for this email in organization
    OrganizationInvite.objects.filter(
        organization=self.organization, target_email__iexact=self.target_email
    ).delete()
```

### **User.join() Method - Membership Creation**

**Location:** [posthog/models/user.py#L347-L391](https://github.com/PostHog/posthog/blob/master/posthog/models/user.py#L347-L391)

```
def join(
    self,
    *,
    organization: Organization,
    level: OrganizationMembership.Level = OrganizationMembership.Level.MEMBER,
) -> OrganizationMembership:
    with transaction.atomic():
        # 1. Create membership record
        membership = OrganizationMembership.objects.create(
            user=self, 
            organization=organization, 
            level=level
        )

        # 2. Update user's current organization/team
        self.current_organization = organization
        if not organization.is_feature_available(AvailableFeature.ADVANCED_PERMISSIONS):
            # Without project-level permissions, set first team
            self.current_team = organization.teams.order_by("id").first()
        else:
            # With project-level permissions, use UserAccessControl to find accessible team
            uac = UserAccessControl(user=self, organization_id=str(organization.id))
            self.current_team = uac.filter_queryset_by_access_level(
                organization.teams.all(), 
                include_all_if_admin=True
            ).order_by("id").first()
        self.save()

        # 3. Auto-assign default role (if organization has configured one)
        if organization.default_role_id:
            try:
                RoleMembership.objects.create(
                    role_id=organization.default_role_id, 
                    user=self, 
                    organization_member=membership
                )
            except Exception as e:
                capture_exception(e, {...})

        # 4. Update billing
        self.update_billing_organization_users(organization)
        
    return membership
```

@shared_task(**EMAIL_TASK_KWARGS)
def send_invite(invite_id: str) -> None:
    invite = OrganizationInvite.objects.select_related("created_by", "organization").get(id=invite_id)
    message = EmailMessage(
        campaign_key=f"invite_email_{invite_id}",
        subject=f"{invite.created_by.first_name} invited you to join {invite.organization.name} on PostHog",
        template_name="invite",
        template_context={
            "invite": invite,
            "expiry_date": (timezone.now() + timedelta(days=INVITE_DAYS_VALIDITY)).strftime(...),
            "inviter_first_name": invite.created_by.first_name or "someone",
            "organization_name": invite.organization.name,
            "url": f"{settings.SITE_URL}/signup/{invite_id}",
        },
        reply_to=invite.created_by.email or "",
    )
    message.add_recipient(email=invite.target_email, distinct_id=f"invite_{invite_id}")
    message.send()
```

@shared_task(**EMAIL_TASK_KWARGS)
def send_member_join(invitee_uuid: str, organization_id: str) -> None:
    invitee = User.objects.get(uuid=invitee_uuid)
    organization = Organization.objects.get(id=organization_id)
    
    message = EmailMessage(
        campaign_key=f"member_join_email_org_{organization_id}_user_{invitee_uuid}",
        subject=f"{invitee.first_name} joined you on PostHog",
        template_name="member_join",
        template_context={
            "invitee": invitee,
            "organization": organization,
            "invitee_first_name": invitee.first_name,
            "organization_name": organization.name,
        },
    )
    # Don't send to the new member themselves
    members_to_email = organization.members.exclude(email=invitee.email)
    for user in members_to_email:
        message.add_user_recipient(user)
    message.send()
```

def _sync_user_product_list_for_accessible_teams(self, user: "User") -> None:

### **Error Handling & Validations**

[**OrganizationInvite.validate()**](https://github.com/PostHog/posthog/blob/master/posthog/models/organization_invite.py#L82-L123) checks:

1. **Email mismatch**: Invite for different email → ValidationError
2. **Expired**: Older than INVITE\_DAYS\_VALIDITY (72 hours) → InviteExpiredException
3. **Account exists**: User exists but not in org, non-authenticated invite → Redirect to login
4. **Already member**: User already in org → ValidationError
5. **Email already used**: Another user with same email in org → ValidationError

[**InviteSignupSerializer.validate()**](https://github.com/PostHog/posthog/blob/master/posthog/api/signup.py#L218-L226):

- Requires `first_name` and `password` for new users
- Checks SSO enforcement
