180: 	try {

183: 		const isDirectory = (stat.type & vscode.FileType.Directory) !== 0
184: 
185: 		if (!isFile && !isDirectory) {
186: 			throw new Error(`URI ${uri.toString()} is neither a file nor a directory`)
187: 		}
188: 
189: 		return isDirectory ? 'directory' : 'file'
190: 	} catch (error) {
191: 		if (error instanceof vscode.FileSystemError && error.code === 'FileNotFound') {
192: 			return null
193: 		}
194: 		throw error
195: 	}
